import sqlite3
import pandas as pd # The super-reader tool

# 1. Read the big list from the internet
print("üåç Reading internet data...")
df = pd.read_csv("https://raw.githubusercontent.com/ilyankou/passport-index-dataset/master/passport-index-tidy-iso2.csv")

# 2. Open the notebook
conn = sqlite3.connect('tara_migration.db')
cursor = conn.cursor()

# 3. Write it all down
for index, row in df.iterrows():
    # Writing: Origin, Destination, Activity, Rule, Priority (3)
    cursor.execute("INSERT INTO mobility_logic VALUES (?, ?, ?, ?, ?)", 
                   (row['Passport'], row['Destination'], 'Tourism', row['Requirement'], 3))

conn.commit()
print("‚úÖ TARA has learned 40,000 rules!")
conn.close()
